# Multi-stage build for Crab-Bot
# Stage 1: Builder
FROM python:3.11-slim as builder

# Install system dependencies
RUN apt-get update && apt-get install -y \
    gcc \
    g++ \
    make \
    git \
    wget \
    curl \
    nmap \
    nikto \
    traceroute \
    dnsutils \
    iputils-ping \
    net-tools \
    iproute2 \
    iptables \
    tcpdump \
    openssh-client \
    sqlite3 \
    libpcap-dev \
    && rm -rf /var/lib/apt/lists/*

# Install signal-cli
RUN wget -q https://github.com/AsamK/signal-cli/releases/download/v0.11.11/signal-cli-0.11.11.tar.gz && \
    tar -xf signal-cli-0.11.11.tar.gz && \
    mv signal-cli-0.11.11 /opt/signal-cli && \
    ln -s /opt/signal-cli/bin/signal-cli /usr/local/bin/signal-cli && \
    rm signal-cli-0.11.11.tar.gz

# Set working directory
WORKDIR /app

# Copy requirements
COPY requirements.txt .

# Install Python dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Stage 2: Final
FROM python:3.11-slim

# Install runtime dependencies
RUN apt-get update && apt-get install -y \
    nmap \
    nikto \
    traceroute \
    dnsutils \
    iputils-ping \
    net-tools \
    iproute2 \
    iptables \
    tcpdump \
    openssh-client \
    sqlite3 \
    libpcap0.8 \
    wget \
    curl \
    chromium \
    chromium-driver \
    fonts-liberation \
    libasound2 \
    libatk-bridge2.0-0 \
    libatk1.0-0 \
    libcups2 \
    libdbus-1-3 \
    libgbm1 \
    libgtk-3-0 \
    libnspr4 \
    libnss3 \
    libx11-xcb1 \
    libxcb1 \
    libxcomposite1 \
    libxdamage1 \
    libxrandr2 \
    xdg-utils \
    && rm -rf /var/lib/apt/lists/*

# Copy signal-cli from builder
COPY --from=builder /opt/signal-cli /opt/signal-cli
COPY --from=builder /usr/local/bin/signal-cli /usr/local/bin/signal-cli

# Create non-root user
RUN groupadd -r crabot && useradd -r -g crabot crabot && \
    mkdir -p /home/crabot/.crab_bot && \
    chown -R crabot:crabot /home/crabot

# Set working directory
WORKDIR /app

# Copy application files
COPY --chown=crabot:crabot . .

# Copy Python packages from builder
COPY --from=builder /usr/local/lib/python3.11/site-packages /usr/local/lib/python3.11/site-packages

# Create necessary directories
RUN mkdir -p reports scan_results alerts monitoring backups temp scripts \
    phishing_templates phishing_logs captured_credentials ssh_keys ssh_logs \
    traffic_logs nikto_results whatsapp_session && \
    chown -R crabot:crabot .

# Set environment variables
ENV PYTHONUNBUFFERED=1 \
    PYTHONDONTWRITEBYTECODE=1 \
    PATH="/usr/local/bin:${PATH}" \
    CHROME_BIN=/usr/bin/chromium \
    CHROMEDRIVER_PATH=/usr/bin/chromedriver

# Switch to non-root user
USER crabot

# Expose ports
EXPOSE 8080 8081 8082

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD python -c "import socket; s=socket.socket(); s.connect(('localhost', 8080))" || exit 1

# Entry point
ENTRYPOINT ["python", "crab_bot.py"]
